рвение
.
если
мы
хоть
отчасти
принести
польза
и
убедить
подобный
род
человек
возненавидеть
сам
себя
и
возлюбить
противоположный
образ
мысль
,
то
наш
вступление
к
закон
о
нечестие
делать
удачно
.
по
крайний
мера
можно
на
это
надеяться
.
если
же
и
нет
,
то
все
равный
подобный
род
рассуждение
нельзя
ставить
в
упрек
законодатель
.
вслед
за
вступление
мы
мочь
выставить
требование
,
правильно
истолковывать
закон
и
повелевать
нечестивый
человек
переменить
свой
образ
жизнь
на
благочестивый
.
для
ослушник
же
пусть
быть
следующий
закон
о
нечестие
:
если
кто-либо
сказать
или
делать

нечестивый
,
любой
присутствовать
должен
это
воспротивиться
и
донести
о
этот
должностной
лицо
,
а
из
этот
тот
,
кто
первый
узнать
о
это
,
должен
,
согласный
с
закон
,
привести
виновный
в
суд
,
назначить
для
разбор
такой
род
дело
.
если
же
какой-нибудь
должностной
лицо
,
быть
извещенный
,
не
делать
это
,
всякий
желать
заступиться
за
закон
мочь
привлечь
этот
лицо
к
суд
.
если
кто
оказаться
виновный
,
суд
должен
назначить
соответствующий
наказание
за
каждый
отдельный
проступок
нечестивый
человек
.
каждый
виновный
должен
быть
заключить
в
тюрьма
,
который
в
государство
быть
три
:
один
–
на
площадь
,
общий
для
большинство
задержать
,
предназначить
для
охрана
личный
безопасность
большинство
;
другой
тюрьма
–
невдалеке
от
место
заседание
ночной
собрание
–
быть
называться
софронистерий
;
третий
же
–
посреди
страна
,
в
какой-нибудь
пустынный
и
совершенно
дикий
место
–
получить
имя
,
который
быть
выражать
возмездие
.
так
как
человек
становиться
нечестивый
по
три
упомянуть
мы
причина
и
каждый
такой
причина
создавать
два
род
нечестивец
,
то
все
получиться
бы
шесть
различный
род
человек
,
заблуждаться
по
повод
бог
.
весь
они
должен
подвергнуться
различный
и
неравный
наказание
.
ибо
тот
,
кто
совершенно
отрицать
бытие
бог
,
но
от
природа
обладать
справедливый
характер
,
ненавидеть
дурной
человек
и
из-за
глубокий
отвращение
к
несправедливость
не
склонный
к
совершение
несправедливый
поступок
,
избегать
человек
несправедливый
,
а
справедливый
любить
.
с
другой
же
род
–
это
тот
,
у
кто
к
мнение
,
будто
Вселенная
лишить
бог
,
добавлять
невоздержанность
в
удовольствие
и
страдание
,
хотя
они
и
обладать
сильный
память
и
прекрасный
восприимчивость
к
наука
.
общий
болезнь
тот
и
другой
тот
,
что
они
не
признавать
бог
;
но
первый
творить
мало
зло
на
пагуба
остальной
человек
,
чем
второй
.
они
в
свой
речь
преисполнить
дерзость
в
отношение
бог
,
жертвоприношение
,
клятва
и
смеяться
также
надо
все
остальной
;
быть
мочь
,
они
и
другой
человек
делать
бы
такой
же
,
если
бы
они
не
настигать
вовремя
правосудие
.
второй
держаться
тот
же
мнение
,
что
и
первый
,
слывуть
за
человек
одаренный
,
но
исполненный
коварство
и
злокозненный
.
из
этот
род
человек
выходить
много
прорицатель
,
человек
,
заниматься
всевозможный
ворожба
,
а
иногда
и
тиран
,
демагог
,
военачальник
,
основатель
частный
таинство
,
а
также
и
изощрить
так
называть
софист
.
разновидность
подобный
род
человек

,
но
особый
законоположение
достойный
два
из
они
:
тот
,
кто
принадлежать
к
один
из
они
,
а
именно
лицемера
,
заслуживать
смертный
казнь
,
и
не
один
и
даже
не
два
,
а
сразу
много
;
человек
же
второй
разновидность
нуждаться
в
увещание
и
тюремный
заключение
.
точно
так
же
и
взгляд
,
отрицать
промысл
бог
,
порождать
два
род
человек
;
человек
,
придерживаться
мнение
,
будто
бог
можно
умолять
,
тоже
два
род
.
ввиду
существовать
между
они
разница
судья
,
опираться
на
закон
,
должен
присудить
тот
,
кто
впасть
в
нечестие
по
неразумие
,
а
не
по
злой
побуждение
и
нрав
,
к
заключение
в
софронистерия
не
менее
чем
на
пять
год
.
в
течение
этот
время
никто
из
гражданин
не
должен
иметь
к
они
доступ
,
кроме
участник
ночной
собрание
,
который
быть
он
увещевать
и
беседовать
с

.
когда
же

срок

,
тот
из

,

себя

,

с

.

же

,
то

если

снова

подобный

,
он
следовать
карать

.
то

,

,
кроме
то
что
не
признавать
бог
и
они
мысть
или

они
умолить
,
вдобавок

животное
и
,
презирать
человек
,
обольщать
некоторый
из
они
при
жизнь
,
уверять
,
будто
мочь
вызывать
душа

,
или
,
обещать

бог

,

,

отдельный

,
так
и

семья
и

,
–
они
,

виновный

,

назначить

в
тюрьма
,
находиться

страна
.

не

к

.
Назначеть
он

.

он

.

из

,
то

мочь
привлечь
он
к

нечестие
.
если
он

себя
ребенок
,
полезный

,
то

попечитель
о

о

как
о

–

,

,

,

.

,

,

.

.

.

,

,

,

.

,

,

,

,

,

,
