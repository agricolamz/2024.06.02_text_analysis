,

,

;

,

брать

у

–

,

.

едва

,

ее

:

ежедневно

.

,

,

стать

,

,

,

)
,

,

.

,

в

,

,

.

весь

,

надо
соблюдать
следующий
.
богиня
милостиво
даровать
мы
два

:

,
виноград
–
недолговечный

,

,

,

.

такой

:

,

ли

,

Арктура
)
,

,

,

,

,

,

;

.

,

,

,

,

.

,

в

.
Метек

,

,

,

.

,

,

,

,

,

и

,

.

,

рвать

.

,

закон

,

,

,

,

,

,

.

,

,

на

,

.

,

.

.
Чужеземец

,

.

,

,

,

.
Ослушник

,

,

оыть

.

,

.

,

,

,

на
то
,

,

;

,

в
сторона
,

,
а
с
вода
весь
это
можно

,
таков
ее
природа
.
поэтому

должен
прийти
на
помощь
закон
.
так

,
пусть

действовать
следующий

о

:
если

чужой
вода
,

ли

,

,

,

,
то

,

он

.

,

портить

,

.

,

.

,

,

.

,

,

.

,

,

.

,

.

.

.

,

назначать
наказание
,

,

,

–

ли

,

,
–

в
тот
же
род
.

это

.

.

установить

,

.

они

,

,

,

.

,

.

:

из

,

.
дело
в
то
,

,
который

нуждаться
в
упражнение
и

,

общегосударственный

.

,
между

.
Усердно

,

,

,

,

.

.

не
должен
одновременно
заниматься
и
плотничий

;

и

заботиться
о
чужой

он

,

о
свой
собственный

,

тот

,

,

на
он

,

,

,

о

,

от

доход
от
его
собственный

.
нет
,

только

,

.

.

,

,

,

,

исправить

.

,

,

человек
,

.

,

,

,

.

,

,
общий
суд
.
никто

,

,

.

,

,

,

,

,

,

,

.

,

.

;

–

,

.

,

,

,

,

,

,

,
или

,

,

или

.
стражи

и

.
но

,

,
по
весь
наш

,

.

,

,

,

.
весь

и

.

,

,

,

,

,

,

,

,

,
второй
–
для
их

,
третий
–
для

,
как

,
что
поселиться
у
мы

в

,

.

,
