однако
я
,
напрягш
память
,
сказать
:
–
Клянусь
Зевс
,
мы
едва
не
пропустить
большой
из
благо
!
–
какой
же
именно
?
–
спросить
он
.
–
счастье
,
мой
Клиний
.
ведь
все
,
даже
самый
неумный
человек
,
утверждать
,
что
это
–
большой
благо
.
–
ты
прав
,
–
сказать
он
.
подумать
еще
раз
,
я
добавить
:
–
а
ведь
мы
едва
не
оказаться
в
смешной
положение
перед
гость
–
и
я
и
ты
,
сын
Аксиох
.
–
какой
это
образ
?
–
спросить
он
.
–
а
такой
,
что
,
назвать
ранее
благо
счастье
,
мы
сейчас
снова
заговорить
о
это
.
–
да
как
же
так
?
–
смешной
ведь
,
когда
положение
,
однажды
уже
выдвинуть
,
повторять
снова
и
дважды
утверждать
один
и
тот
же
.
–
что
,
–
спрашивать
он
,
–
ты
иметь
в
вид
?
–
да
ведь
мудрость
,
–
отвечать
я
,
–
и
быть
счастье
,
это
известный
любой
ребенок
.
Клиния
выразить
удивление
:
настолько
он
еще
молодой
и
простодушный
.
я
же
,
видеть
он
изумление
,
сказать
:
–
разве
ты
неведомый
,
Клиний
,
что
в
мастерство
игра
на
флейт
весь
счастливый
флейтист
?
Клиний
согласиться
.
–
а
разве
в
дело
написание
и
прочтение
текст
не
весь
счастливый
грамматика
?
–
разумеется
.
–
ну
а
что
касаться
опасность
в
море
,
неужели
ты
думать
,
что
быть
кто-либо
счастливый
в
избежание
они
,
что
,
как
правило
,
искусный
кормчий
?
–
нет
,
конечно
.
–
ну
а
отправляться
по
война
,
с
кто
бы
ты
охотно
разделить
опасность
и
успех
–
с
искусный
стратег
или
с
невежественный
?
–
с
искусный
.
–
а
более
,
кто
бы
ты
рискнуть
доверить
–
искусный
врач
или
неуча
?
–
Искусный
.
–
следовательно
,
–
сказать
я
,
–
ты
считать
,
что
больше
счастье
приносить
сотрудничество
с
мудрый
человек
,
чем
с
неучто
?
он
согласиться
.
–
значит
,
мудрость
в
весь
нести
человек
счастье
,
ибо
мудрость
ни
в
что
не
ошибаться
,
но
необходимый

правильно

и
преуспевать
.
иначе
она

и
не
быть

мудростий
.
в
конец

мы
как-то
так
согласиться
,
что
в
целое
,
если
у
кто
налицо

,
тот

нуждаться

другой

.
а
прийти
с
он
к
такой
соглашение
,
я
возобновить
свой

относительно
наш

,
как
обстоять
с
она
дело
.
–
мы
согласиться
,
–
сказать
я
,
–
что
если
у
мы
быть
весь
благо
,
то
мы
то
самый

?

это
.
–
ну
а
счастливый

,
если
они
приносить
мы
польза
или
если
не

?
–
если
приносить
,
–
отвечать
он
.
–
а
приносить
они
мы
польза
лишь
в
сила
обладание
они
,
даже
если
мы
они
не
пользоваться
?

,
если
бы

мы

,
но
мы

не
ель
,

много
питье
,
но
мы
бы
не
пить
,
–
принести

это

?
–
конечно
,
нет
,
–
отвечать
он
.
–
далеко
:
если
бы
весь
мастер
иметь
в
свой
распоряжение
все
необходимый
для
свой

,
но
не
пользоваться
это
,
быть
бы
они
счастливый
в
сила
сам
обладание
то
,
что
нужный
мастер
?
например
,
если
бы
плотник
иметь
в
свой
распоряжение
весь
необходимый

и

,
но
не
работать
бы
с
они
,
извлечь
бы

?
–
ни

кой

,
–
отвечать
Клиний
.
–
ну
а
если
бы
кто-нибудь
обрести
богатство

весь

,
о

сейчас
говорить
,
но
не
пользоваться
бы
весь
это
,
быть
бы
он

?
–
конечно
,

,
Сократ
.
–
следовательно
,

,

.
в

.
–
истинный

.
–
так
,
значить
.

,

,

,

?
–
я

,

.
–

,
–
спросить
я
,
–
или

?
–
нет
,

.
–

,
–
промолвить
я
.
–

,

,

,

–

.

?
он

.
–

,
разве

,

?
–
нет
,

,
–

Клиний
.
–

?

.
–

ведь
.
–

я
,
–

,

.

.

.
